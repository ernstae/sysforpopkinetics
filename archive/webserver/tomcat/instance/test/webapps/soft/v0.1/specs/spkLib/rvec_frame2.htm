Description
Reads the original matrix in the row-major order and returns as a column vector
                                                      T
    [ x   , x    , ... , x  ,  x   ,  x   , ... , x  ]
       1,1   1,2          1,n   2,1    2,2         m,n
where m is the number of rows and n is the number of columns in the matrix X.

Example
If you compile, link and run the following program,

  #include <iostream>
  #include "DoubleMatrix.h"
  #include "rvec.h"

  void main(){
    
    using namespace std;

    DoubleMatrix dmatA( 2, 2 );
    double * pDmatA = dmatA.data();

    DoubleMatrix dmatB;

    pDmatA[0] = 1.0;
    pDmatA[1] = 3.0;
    pDmatA[2] = 2.0;
    pDmatA[3] = 4.0;

    dmatB = rvec( dmatA );
    
    cout << "dmatA =" << endl;
    dmatA.print();
    cout << "rvec(dmatA) =" << endl;
    dmatB.print();
  }
 the following results will be displayed.

  dmatA = 
  [ 1  2 ]
  [ 3  4 ]

  rvec(dmatA) =
  [ 1 ]
  [ 2 ]
  [ 3 ]
  [ 4 ]
