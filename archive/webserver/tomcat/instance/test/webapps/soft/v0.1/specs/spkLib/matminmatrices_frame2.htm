Description
Given m by n matrices A and B, matmin(A, B) returns another m by n matrix C such that c(i,j) = a(i,j) if a(i,j) <= b(i,j), or else c(i,j) = b(i,j) if b(i,j) < a(i,j), for all i and j such that m > i > 0 and n > j > 0.

Example
If you compile, link and run the following program,

    #include <iostream>
    #include "matmin.h"

    void main(){

      using namespace std;

      DoubleMatrix dmatA(3,2),
                   dmatB(3,2),
                   dmatC(3,2);
      double       *pdA    = dmatA.data(),
                   *pdB    = dmatB.data();

      // Setting A to a matrix:
      //    [  1  -4 ]
      //    [ -2   5 ]
      //    [  3  -6 ]
      // and B to a matrix:
      //    [  0  -3 ]
      //    [ -1   4 ]
      //    [  2  -5 ]
      for( int i=0; i<dmatA.nr()*dmatA.nc(); i++ ){
          pdA[i] = (i+1)*(i%2 == 0? 1: -1);
          pdB[i] = ( i )*(i%2 == 0? 1: -1);
      }

      dmatC = matmin(dmatA, dmatB);

      cout << "Original matrix A" << endl;
      dmatA.print();

      cout << "Original matrix B" << endl;
      dmatB.print();

      cout << "Matrix that contains smallers of A and B" << endl;
      dmatC.print();
    }

 following matrices are displayed.
    Original matrix A
    [1, -4]
    [-2, 5]
    [3, -6]
    Original matrix B
    [0, -3]
    [-1, 4]
    [2, -5]
    Matrix that contains smallers of A and B
    [0, -4]
    [-2, 4]
    [2, -6]
