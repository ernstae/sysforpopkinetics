Description
Returns the matrix inverse of the square, symmetric and positive-definite matrix, a, stored in a valarray object in the column order.

Arguments
dmatA
The valarray a contains the square, symmetric, positive-definite matrix A in the column major order. If the matrix did not have these properties, it throws an SpkException .

Example
If you compile and link the C++ program,

#include <iostream>
#include <iomanip>
#include "SpkValarray.h"
#include "inverse.h"

void main()
{
  using namespace std;

  int nRows = 2;
  int nCols= nARow;
  valarray<double> a (nRows * nCols);

  a[0 + 0 * nARow] = 1000000.0;
  a[0 + 1 * nARow] =      -1.0;
  a[1 + 0 * nARow] =      -1.0;
  a[1 + 1 * nARow] =       2.0;
  
  cout << setiosflags(ios::scientific) << setprecision(15);

  cout << "a = " << endl;
  DoubleMatrix( a, nRows ).print();
  cout << endl;

  cout << "inverse(a, nCols) = " << endl;
  DoubleMatrix( inverse(a, nCols), nCols ).print();
  cout << endl;
}
