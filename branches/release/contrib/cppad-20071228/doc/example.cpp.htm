namespace {
     // function that runs one test
     static size_t Run_ok_count    = 0;
     static size_t Run_error_count = 0;
     bool Run(bool TestOk(void), const char *name)
     {    bool ok = true;
          ok &= TestOk();
          if( ok )
          {    std::cout << "Ok:    " << name << std::endl;
               Run_ok_count++;
          }
          else
          {    std::cout << "Error: " << name << std::endl;
               Run_error_count++;
          }
          return ok;
     }
}
