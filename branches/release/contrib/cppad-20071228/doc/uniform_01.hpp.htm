# include <cstdlib>

namespace CppAD {
     inline void uniform_01(size_t seed)
     {    std::srand( (unsigned int) seed); }

     template <class Vector>
     void uniform_01(size_t n, Vector &x)
     {    static double factor = 1. / double(RAND_MAX);
          while(n--)
               x[n] = std::rand() * factor;
     }
