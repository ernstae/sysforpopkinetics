Matrix of Zeros

Syntax
Matlab or Octave x = zeros(m, n)
C++ zero_matrix<double> x(m, n)

Matlab or Octave
If m and n are integer scalars, the Matlab or Octave syntax
     x = zeros(m, n)
sets x to an  m \times n matrix with all its elements equal to zero.

Example
     function [ok] = zeros_ok()
     ok      = true;
     m       = 2;
     n       = 3;
     % --------------------
     x       = zeros(2, 3);
     % --------------------
     [m, n]  = size(x);
     ok      = ok & (m == 2);
     ok      = ok & (n == 3);
     ok      = ok & all( all( x == 0 ) );
     return

C++
The corresponding C++ syntax is
     zero_matrix<double> x(m, n)
where m and n are size_t objects and x is an  m \times n ublas matrix<double> object.

Example
     # include <mat2cpp.hpp>
     bool zeros_ok(void)
     {    bool   ok  = true;
          using namespace mat2cpp;

          size_t i, j, m(2), n(3);
          // -------------------------
          zero_matrix<double> x(m, n);
          // -------------------------
          ok &= (x.size1() == m);
          ok &= (x.size2() == n);
          for(i = 0; i < m; i++)
          {    for(j = 0; j < n; j++)
                    ok &= (x(i,j) == 0.);
          }
          return ok;
     }
