Description
Print the elements of the matrix proceeded by the dimensions to STDIO. If the matrix is empty, it prints nothing.

Example
If you compile, link and run the following program
#include <iostream>
#include "DoubleMatrix.h"

void main(){

     using namespace std;

     DoubleMatrix A(3, 2);
     double *pA = A.data();
     DoubleMatrix Empty;

     // Initialize A to a matrix
     // [ 1  4 ]
     // [ 2  5 ]
     // [ 3  6 ]
     for( int i=0; i<6; i++ )
          pA[i] = i+1;

     cout << "A = " << endl;
     A.print();
     cout << endl;
     cout << "empty matrix = " << endl;
     Empty.print();
}

 a matrix  
     A =
     [ 1 4 ]
     [ 2 5 ]
     [ 3 6 ]

     empty matrix =
    []

will be printed.
