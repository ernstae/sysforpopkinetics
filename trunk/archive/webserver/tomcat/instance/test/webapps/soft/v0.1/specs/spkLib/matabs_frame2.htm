Description
Given a m by n matrix A, matabs(A) returns another m by n matrix B, such that (b(i,j) in B) = (|a(i,j)| in A), where m > i > 0 and n > j > 0.

Example
If you compile, link and run the following program,
#include <iostream>
#include "matabs.h"

void main(){

  using namespace std;

  DoubleMatrix dmatA(3,2), dmatB(3,2);
  double       *pdA = dmatA.data();

  // Setting A to a matrix:
  //    [  1  -4 ]
  //    [ -2   5 ]
  //    [  3  -6 ]
  for( int i=0; i<dmatA.nr()*dmatA.nc(); i++ ){
      pdA[i] = (i+1)*(i%2 == 0.0? 1.0: -1.0);
  }

  dmatB = matabs(dmatA);

  cout << "Original matrix A" << endl;
  dmatA.print();

  cout << "Absolute version of A" << endl;
  dmatB.print();

 a following matrix is displayed.
Original matrix A
  [  1  -4 ]
  [ -2   5 ]
  [  3  -6 ]
Absolute version of A
  [ 1  4 ]
  [ 2  5 ]
  [ 3  6 ]
